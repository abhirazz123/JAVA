		x = 10;
		y = x++;
	}
	public Demo(int x, int y) {
		this.x += this.x + x++;
		this.y -= this.x + --y;
	}
}
public class Test {
	public static void main(String[] args) {
		System.out.print(new Demo().x + new Demo().y);
